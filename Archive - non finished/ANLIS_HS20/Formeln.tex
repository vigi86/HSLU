\section*{Formelsammlung}
\subsection*{Einfache lineare Funktionen}
Lineare Funktion
\begin{align*}
    y=&ax+b \\
    y_2-y_1=&a(x_2-x_1)\\
    a=&\frac{y_2-y_1}{x_2-x_1}=\frac{\Delta y}{\Delta x}
\end{align*}

Differenzenquotient
\begin{align*}
    \frac{\Delta y}{\Delta x}=\frac{f(x+\Delta x)-f(x)}{\Delta x}
\end{align*}
